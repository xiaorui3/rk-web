2.类型转化的问题
	自动类型转化
		byte a = 1;		1--32bit		8bit
	强制类型转化
		int a = 1;		32bit	1
		byte b = (byte)a;	8bit		127
		b===1

	分析：哪些类型之间能转化
		byte a = 1;			自动转化
		int b = a;		ok

		int a = 1;
		byte b = a;	no		需要强制	byte b = (byte)a;

		float a = 3.4F;			相当于做了个强制	F
		double b = a;	ok

		double a = 3.4;
		float b = a;	no		需要强制	float b = (float)a;
